97 void update_feedback_piezo() {
98  if (feedbackPiezoMode > 0) {
99  if (feedbackPiezoMode == KG_PIEZO_MODE_LONGBEEP && feedbackPiezoTick % 50 == 0) { feedback_set_piezo_tone(feedbackPiezoTick % 100 >= 50 ? 0 : feedbackPiezoFrequency); }
100  else if (feedbackPiezoMode == KG_PIEZO_MODE_LONGPULSE && feedbackPiezoTick % 25 == 0) { feedback_set_piezo_tone(feedbackPiezoTick % 100 >= 25 ? 0 : feedbackPiezoFrequency); }
101  else if (feedbackPiezoMode == KG_PIEZO_MODE_SHORTBEEP && feedbackPiezoTick % 10 == 0) { feedback_set_piezo_tone(feedbackPiezoTick % 20 >= 10 ? 0 : feedbackPiezoFrequency); }
102  else if (feedbackPiezoMode == KG_PIEZO_MODE_SHORTPULSE && feedbackPiezoTick % 5 == 0) { feedback_set_piezo_tone(feedbackPiezoTick % 20 >= 5 ? 0 : feedbackPiezoFrequency); }
103  else if (feedbackPiezoMode == KG_PIEZO_MODE_TINYBEEP && feedbackPiezoTick % 5 == 0) { feedback_set_piezo_tone(feedbackPiezoTick % 10 >= 5 ? 0 : feedbackPiezoFrequency); }
104  else if (feedbackPiezoMode == KG_PIEZO_MODE_TINYPULSE && feedbackPiezoTick % 2 == 0) { feedback_set_piezo_tone(feedbackPiezoTick % 10 >= 2 ? 0 : feedbackPiezoFrequency); }
105  if (++feedbackPiezoTick >= feedbackPiezoDuration && feedbackPiezoDuration > 0) feedback_set_piezo_mode(KG_PIEZO_MODE_OFF, 0, 0);
106  }
107 }

137 uint16_t kg_cmd_feedback_set_piezo_mode(uint8_t index, uint8_t mode, uint8_t duration, uint16_t frequency) {
138  if (mode >= KG_PIEZO_MODE_MAX) {
139  return KG_PROTOCOL_ERROR_PARAMETER_RANGE;
140  } else {
141  // "index" is currently ignored, as there is only one piezo device in the design
142  feedback_set_piezo_mode((feedback_piezo_mode_t)mode, duration, frequency);
143 
144  // send kg_evt_feedback_piezo_mode packet (if we aren't setting it from an API command)
145  if (!inBinPacket) {
146  uint8_t payload[5] = { index, mode, duration, frequency & 0xFF, frequency >> 8 };
147  skipPacket = 0;
148  if (kg_evt_feedback_piezo_mode) skipPacket = kg_evt_feedback_piezo_mode(index, mode, duration, frequency);
149  if (!skipPacket) send_keyglove_packet(KG_PACKET_TYPE_EVENT, 5, KG_PACKET_CLASS_FEEDBACK, KG_PACKET_ID_EVT_FEEDBACK_PIEZO_MODE, payload);
150  }
151  }
152  return 0; // success
153 }
