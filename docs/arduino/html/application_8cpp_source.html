146 uint8_t my_kg_evt_touch_status(uint8_t status_len, uint8_t *status_data) {
147  // XOR current and previous to identify which bits have been changed
148  for (uint8_t i = 0; i < KG_BASE_COMBINATION_BYTES; i++) {
149  touch_data_xor[i] = touch_data_prev[i] ^ status_data[i];
150  }
151 
152  // bits set in "touch_data_prev" indicate which touches are on as of last time any touches changed
153  // bits set in "touch_data_xor" indicate which touches have just flipped between on and off
154  // bits set in "status_data" indicate which touches are on right now
155 
156  // check to see whether this is adding or removing
157  if (memcmp(status_data, touch_data_prev, KG_BASE_COMBINATION_BYTES) > 0) {
158  // touch added
159  if (KGT_AY(touch_data_xor)) kg_cmd_motion_set_mode(0, true); // enable motion sensor 0 (default MPU-6050 on back of hand)
160  else if (KGT_DY(touch_data_xor)) keyboard_key_down(KEY_A); // send key-down report for 'A' key
161  } else {
162  // touch removed
163  if (KGT_AY(touch_data_xor)) kg_cmd_motion_set_mode(0, false); // disable motion sensor 0
164  else if (KGT_DY(touch_data_xor)) keyboard_key_up(KEY_A); // send key-up report for 'A' key
165  else if (KGT_GY(touch_data_xor)) keyboard_key_press(KEY_A); // send key down and key-up report for 'A' key
166  }
167 
168  // store current touch status for comparing next time
169  memcpy(touch_data_prev, status_data, status_len);
170 
171  // allow KGAPI event packet transmission
172  return 0;
173 }
