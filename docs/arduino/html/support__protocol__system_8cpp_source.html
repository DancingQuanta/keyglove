57 uint8_t process_protocol_command_system(uint8_t *rxPacket) {
58  // check for valid command IDs
59  uint8_t protocol_error = 0;
60  switch (rxPacket[3]) {
61  case KG_PACKET_ID_CMD_SYSTEM_PING: // 0x01
62  // system_ping()(uint32_t runtime)
63  // parameters = 0 bytes
64  if (rxPacket[1] != 0) {
65  // incorrect parameter length
66  protocol_error = KG_PROTOCOL_ERROR_PARAMETER_LENGTH;
67  } else {
68  // run command
69  uint32_t runtime;
70  uint16_t result = kg_cmd_system_ping(&runtime);
71 
72  // build response
73  uint8_t payload[4] = { runtime & 0xFF, (runtime >> 8) & 0xFF, (runtime >> 16) & 0xFF, (runtime >> 24) & 0xFF };
74 
75  // send response
76  send_keyglove_packet(KG_PACKET_TYPE_COMMAND, 4, rxPacket[2], rxPacket[3], payload);
77  }
78  break;
79 
80  case KG_PACKET_ID_CMD_SYSTEM_RESET: // 0x02
81  // system_reset(uint8_t type)(uint16_t result)
82  // parameters = 1 byte
83  if (rxPacket[1] != 1) {
84  // incorrect parameter length
85  protocol_error = KG_PROTOCOL_ERROR_PARAMETER_LENGTH;
86  } else {
87  // run command
88  uint16_t result = kg_cmd_system_reset(rxPacket[4]);
89 
90  // build and send response if needed
91  if (result != 0xFFFF) {
92  // build response
93  uint8_t payload[2] = { result & 0xFF, (result >> 8) & 0xFF };
94 
95  // send response
96  send_keyglove_packet(KG_PACKET_TYPE_COMMAND, 2, rxPacket[2], rxPacket[3], payload);
97  }
98  }
99  break;
100 
101  case KG_PACKET_ID_CMD_SYSTEM_GET_INFO: // 0x03
102  // system_get_info()(uint8_t major, uint8_t minor, uint8_t patch, uint32_t timestamp)
103  // parameters = 0 bytes
104  if (rxPacket[1] != 0) {
105  // incorrect parameter length
106  protocol_error = KG_PROTOCOL_ERROR_PARAMETER_LENGTH;
107  } else {
108  // run command
109  uint8_t major;
110  uint8_t minor;
111  uint8_t patch;
112  uint32_t timestamp;
113  uint16_t result = kg_cmd_system_get_info(&major, &minor, &patch, &timestamp);
114 
115  // build response
116  uint8_t payload[7] = { major, minor, patch, timestamp & 0xFF, (timestamp >> 8) & 0xFF, (timestamp >> 16) & 0xFF, (timestamp >> 24) & 0xFF };
117 
118  // send response
119  send_keyglove_packet(KG_PACKET_TYPE_COMMAND, 7, rxPacket[2], rxPacket[3], payload);
120  }
121  break;
122 
123  case KG_PACKET_ID_CMD_SYSTEM_GET_MEMORY: // 0x04
124  // system_get_memory()(uint32_t free_ram, uint32_t total_ram)
125  // parameters = 0 bytes
126  if (rxPacket[1] != 0) {
127  // incorrect parameter length
128  protocol_error = KG_PROTOCOL_ERROR_PARAMETER_LENGTH;
129  } else {
130  // run command
131  uint32_t free_ram;
132  uint32_t total_ram;
133  uint16_t result = kg_cmd_system_get_memory(&free_ram, &total_ram);
134 
135  // build response
136  uint8_t payload[8] = { free_ram & 0xFF, (free_ram >> 8) & 0xFF, (free_ram >> 16) & 0xFF, (free_ram >> 24) & 0xFF, total_ram & 0xFF, (total_ram >> 8) & 0xFF, (total_ram >> 16) & 0xFF, (total_ram >> 24) & 0xFF };
137 
138  // send response
139  send_keyglove_packet(KG_PACKET_TYPE_COMMAND, 8, rxPacket[2], rxPacket[3], payload);
140  }
141  break;
142 
143  case KG_PACKET_ID_CMD_SYSTEM_SET_TIMER: // 0x05
144  // system_set_timer(uint8_t handle, uint16_t interval, uint8_t oneshot)(uint16_t result)
145  // parameters = 4 bytes
146  if (rxPacket[1] != 4) {
147  // incorrect parameter length
148  protocol_error = KG_PROTOCOL_ERROR_PARAMETER_LENGTH;
149  } else {
150  // run command

152

154  uint8_t payload[2] = { result & 0xFF, (result >> 8) & 0xFF };
155 
156  // send response
157  send_keyglove_packet(KG_PACKET_TYPE_COMMAND, 2, rxPacket[2], rxPacket[3], payload);
158  }
159  break;
160 
161  case KG_PACKET_ID_CMD_SYSTEM_GET_BATTERY_STATUS: // 0x06
162  // system_get_battery_status()(uint8_t status, uint8_t level)
163  // parameters = 0 bytes
164  if (rxPacket[1] != 0) {
165  // incorrect parameter length
166  protocol_error = KG_PROTOCOL_ERROR_PARAMETER_LENGTH;
167  } else {
168  // run command

172 
173  // build response
174  uint8_t payload[2] = { status, level };
175 
176  // send response
177  send_keyglove_packet(KG_PACKET_TYPE_COMMAND, 2, rxPacket[2], rxPacket[3], payload);
178  }
179  break;
180 
181  default:
182  protocol_error = KG_PROTOCOL_ERROR_INVALID_COMMAND;
183  }
184  return protocol_error;
185 }
