338 uint16_t kg_cmd_system_reset(uint8_t type) {
339  // validate type
340  if (type < 1 || type > 2) {
341  return KG_PROTOCOL_ERROR_PARAMETER_RANGE;
342  }
343 
344  // send response
345  uint8_t payload[] = { 0x00, 0x00 };
346  send_keyglove_packet(KG_PACKET_TYPE_COMMAND, sizeof(payload), KG_PACKET_CLASS_SYSTEM, KG_PACKET_ID_CMD_SYSTEM_RESET, payload);
347 
348  // reboot Keyglove
349  systemResetFlags = 0xFF; // reset everything
350  reset_keyglove_rx_packet(); // clear packet status
351  if (type == KG_SYSTEM_RESET_TYPE_KGONLY) {
352  systemResetFlags = 0; // don't reset anything except the main core system
353  }
354  setup();
355  return 0xFFFF; // success, but DON'T send response packet (already done)
356 }

396 uint16_t kg_cmd_system_set_timer(uint8_t handle, uint16_t interval, uint8_t oneshot) {
397  if (handle > 7) {
398  return KG_PROTOCOL_ERROR_PARAMETER_RANGE;
399  }
400  if (interval == 0) {
401  // stop this timer
402  keygloveSoftTimers &= ~(1 << handle);
403  } else {
404  // schedule/reschedule this timer
405  keygloveSoftTimers |= (1 << handle);
406  if (oneshot) keygloveSoftTimersRepeat &= ~(1 << handle);
407  else keygloveSoftTimersRepeat |= (1 << handle);
408  keygloveSoftTimerInterval[handle] = interval;
409  keygloveSoftTimerSec[handle] = keygloveTock + (interval / 100);
410  keygloveSoftTimer10ms[handle] = keygloveTick + (interval % 100);
411  if (keygloveSoftTimer10ms[handle] > 99) {
412  keygloveSoftTimer10ms[handle] -= 100;
413  keygloveSoftTimerSec[handle]++;
414  }
415  }
416  return 0; // success
417 }
