85  touchTime = millis();
86  } else if (memcmp(touches_verify, touches_active, KG_BASE_COMBINATION_BYTES) != 0 && millis() - touchTime >= opt_touch_detect_threshold) {
87  // detection is over threshold and current readings are different from previous readings
88 
89  #ifdef DEBUG_TOUCH

98  // set official sensor readings to current readings
99  memcpy(touches_active, touches_verify, KG_BASE_COMBINATION_BYTES);

102  uint8_t payload[KG_BASE_COMBINATION_BYTES + 1];

184 void touch_toggle_mode(uint8_t mode) {
185  DEBUG_TOUCHSET(Serial.print("touchset touch_toggle_mode "));
186  DEBUG_TOUCHSET(Serial.println(mode));
187 
188  // find the mode and disable it if it's in the stack
189  uint8_t i;
190  for (i = 0; i < touchModeStackPos && touchModeStack[i] != mode; i++);
191  if (i < touchModeStackPos) {
192  // enabled, so turn it off
193  //deactivate_mode(mode);
194  for (i++; i < touchModeStackPos; i++) touchModeStack[i - 1] = touchModeStack[i];
195  touchModeStackPos--;
196  } else {
197  // not enabled, so turn it on
198  touch_push_mode(mode);
199  }
200 }
