65 void keyboard_key_down(uint8_t code) {
66  DEBUG_TOUCHSET(Serial.print("touchset keydown "));
67  DEBUG_TOUCHSET(Serial.println(code));
68  #if KG_HID & KG_HID_KEYBOARD
69  uint8_t usePos = 0;
70  for (usePos = 0; usePos < 6 && hidKeysDown[usePos] != 0; usePos++);
71  if (usePos == 6) return; // out of HID keyboard buffer space, REALLY weird for the Keyglove!
72  hidKeysDown[usePos] = code;
73  switch (usePos) {
74  case 0:
75  #if KG_HOSTIF & KG_HOSTIF_USB_HID
76  Keyboard.set_key1(code);
77  #endif /* ENABLE_USB */
78  #if KG_HOSTIF & KG_HOSTIF_BT2_HID
79  BTKeyboard.set_key1(code);
80  #endif /* ENABLE_BLUETOOTH */
81  break;
82  case 1:
83  #if KG_HOSTIF & KG_HOSTIF_USB_HID
84  Keyboard.set_key2(code);
85  #endif /* ENABLE_USB */
86  #if KG_HOSTIF & KG_HOSTIF_BT2_HID
87  BTKeyboard.set_key2(code);
88  #endif /* ENABLE_BLUETOOTH */
89  break;
90  case 2:
91  #if KG_HOSTIF & KG_HOSTIF_USB_HID
92  Keyboard.set_key3(code);
93  #endif /* ENABLE_USB */
94  #if KG_HOSTIF & KG_HOSTIF_BT2_HID
95  BTKeyboard.set_key3(code);
96  #endif /* ENABLE_BLUETOOTH */
97  break;
98  case 3:
99  #if KG_HOSTIF & KG_HOSTIF_USB_HID
100  Keyboard.set_key4(code);
101  #endif /* ENABLE_USB */
102  #if KG_HOSTIF & KG_HOSTIF_BT2_HID

105  break;

107  #if KG_HOSTIF & KG_HOSTIF_USB_HID
