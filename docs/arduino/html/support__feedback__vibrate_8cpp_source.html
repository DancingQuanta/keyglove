89 void update_feedback_vibrate() {
90  if (feedbackVibrateMode > 0) {
91  if (feedbackVibrateMode == KG_VIBRATE_MODE_LONGBUZZ && feedbackVibrateTick % 50 == 0) feedback_set_vibrate_logic(feedbackVibrateTick % 100 >= 50 ? 0 : 1);
92  else if (feedbackVibrateMode == KG_VIBRATE_MODE_LONGPULSE && feedbackVibrateTick % 25 == 0) feedback_set_vibrate_logic(feedbackVibrateTick % 100 >= 25 ? 0 : 1);
93  else if (feedbackVibrateMode == KG_VIBRATE_MODE_SHORTBUZZ && feedbackVibrateTick % 10 == 0) feedback_set_vibrate_logic(feedbackVibrateTick % 20 >= 10 ? 0 : 1);
94  else if (feedbackVibrateMode == KG_VIBRATE_MODE_SHORTPULSE && feedbackVibrateTick % 5 == 0) feedback_set_vibrate_logic(feedbackVibrateTick % 20 >= 5 ? 0 : 1);
95  else if (feedbackVibrateMode == KG_VIBRATE_MODE_TINYBUZZ && feedbackVibrateTick % 5 == 0) feedback_set_vibrate_logic(feedbackVibrateTick % 10 >= 5 ? 0 : 1);
96  if (++feedbackVibrateTick >= feedbackVibrateDuration && feedbackVibrateDuration > 0) feedback_set_vibrate_mode(KG_VIBRATE_MODE_OFF, 0);
97  }
98 }

125 uint16_t kg_cmd_feedback_set_vibrate_mode(uint8_t index, uint8_t mode, uint8_t duration) {
126  if (mode >= KG_VIBRATE_MODE_MAX) {
127  return KG_PROTOCOL_ERROR_PARAMETER_RANGE;
128  } else {
129  // "index" is currently ignored, as there is only one vibration device in the design
130  feedback_set_vibrate_mode((feedback_vibrate_mode_t)mode, duration);
131 
132  // send kg_evt_feedback_vibrate_mode packet (if we aren't setting it from an API command)
133  if (!inBinPacket) {
134  uint8_t payload[3] = { index, mode, duration };
135  skipPacket = 0;
136  if (kg_evt_feedback_piezo_mode) skipPacket = kg_evt_feedback_vibrate_mode(index, mode, duration);
137  if (!skipPacket) send_keyglove_packet(KG_PACKET_TYPE_EVENT, 3, KG_PACKET_CLASS_FEEDBACK, KG_PACKET_ID_EVT_FEEDBACK_PIEZO_MODE, payload);
138  }
139  }
140  return 0; // success
