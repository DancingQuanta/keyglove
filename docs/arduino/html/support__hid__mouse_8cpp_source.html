79 void update_hid_mouse() {
80  #if KG_MOTION > 0
81  switch (opt_hid_mouse_mode) {
82  case MOUSE_MODE_TILT_VELOCITY:
83  hidMouseDX += ((gv.y < 0) ? -pow(-(float)gv.y/30, 1.3)*3 : pow((float)gv.y/30, 1.3)*3) - ((gv.z < 0) ? -pow(-(float)gv.z/30, 1.3)*3 : pow((float)gv.z/30, 1.3)*3);
84  hidMouseDY += (gv.x < 0) ? -pow(-(float)gv.x/30, 1.3)*3 : pow((float)gv.x/30, 1.3)*3;
85  break;
86  case MOUSE_MODE_TILT_POSITION:
87  hidMouseDX = ((gv.y < 0) ? -pow(-(float)gv.y/30, 1.3)*3 : pow((float)gv.y/30, 1.3)*3) - ((gv.z < 0) ? -pow(-(float)gv.z/30, 1.3)*3 : pow((float)gv.z/30, 1.3)*3);
88  hidMouseDY = (gv.x < 0) ? -pow(-(float)gv.x/30, 1.3)*3 : pow((float)gv.x/30, 1.3)*3;
89  break;
90  case MOUSE_MODE_MOVEMENT_POSITION:
91  #if (KG_FUSION > 0)
92  hidMouseDX = apFrame.x * opt_hid_mouse_scale_mode4[0];
93  hidMouseDY = apFrame.y * opt_hid_mouse_scale_mode4[1];
94  #else
95  hidMouseDX = aa.x * opt_hid_mouse_scale_mode4[0];
96  hidMouseDY = aa.y * opt_hid_mouse_scale_mode4[1];
97  #endif
98  break;
99  case MOUSE_MODE_3D:
100  #if (KG_FUSION > 0)
101  hidMouseDX = apFrame.x * opt_hid_mouse_scale_mode4[0];
102  hidMouseDY = apFrame.y * opt_hid_mouse_scale_mode4[1];
103  hidMouseDZ = apFrame.z * opt_hid_mouse_scale_mode4[2];
104  #else
105  hidMouseDX = aa.x * opt_hid_mouse_scale_mode4[0];
106  hidMouseDY = aa.y * opt_hid_mouse_scale_mode4[1];
107  hidMouseDZ = aa.z * opt_hid_mouse_scale_mode4[2];
108  #endif
109  break;
110  }
111  switch (opt_hid_scroll_mode) {
112  case SCROLL_MODE_TILT_VELOCITY: // gyro
113  break;
114  case SCROLL_MODE_TILT_POSITION: // gyro
115  hidScrollDY -= ((float)gv.y < 0) ? -sqrt(-(float)gv.y / 30) : sqrt((float)gv.y / 30);
116  break;
117  case SCROLL_MODE_MOVEMENT_POSITION: // accel
118  break;
119  }
120  #endif
121 
122  if (opt_hid_mouse_mode > 0 && (hidMouseDX != 0 || hidMouseDY != 0 || hidMouseDZ != 0)) {
123  if (opt_hid_mouse_invert_x == 1) hidMouseDX = -hidMouseDX;
124  if (opt_hid_mouse_invert_y == 1) hidMouseDY = -hidMouseDY;
125  if (opt_hid_mouse_invert_z == 1) hidMouseDZ = -hidMouseDZ;
126  #if KG_HOSTIF & KG_HOSTIF_USB_HID
127  if (hidMouseDZ > 0) Mouse.move(hidMouseDX, hidMouseDY, hidMouseDZ);
128  else Mouse.move(hidMouseDX, hidMouseDY);
129  #endif /* KG_HOSTIF_USB_HID */
130  #if KG_HOSTIF & KG_HOSTIF_BT2_HID
131  if (hidMouseDZ > 0) BTMouse.move(hidMouseDX, hidMouseDY, hidMouseDZ, 0);
132  else BTMouse.move(hidMouseDX, hidMouseDY);
133  #endif /* KG_HOSTIF_BT2_HID */
134  #if KG_HOSTIF & KG_HOSTIF_R400_HID
135  if (hidMouseDZ > 0) RX400.move(hidMouseDX, hidMouseDY, hidMouseDZ);
136  else RX400.move(hidMouseDX, hidMouseDY);
137  #endif /* KG_HOSTIF_R400_HID */
138  DEBUG_HID_MOUSE(Serial.print("mouse "));
139  DEBUG_HID_MOUSE(Serial.print(hidMouseDX)); DEBUG_HID_MOUSE(Serial.print(" "));
140  DEBUG_HID_MOUSE(Serial.print(hidMouseDY)); DEBUG_HID_MOUSE(Serial.print(" "));
141  DEBUG_HID_MOUSE(Serial.println(hidMouseDZ));
142  }
143  if (opt_hid_scroll_mode > 0 && hidScrollDY != 0) {
144  if (opt_hid_mouse_invert_y == 1) hidMouseDY = -hidMouseDY;
145  #if KG_HOSTIF & KG_HOSTIF_USB_HID
146  Mouse.scroll(hidScrollDY);
147  #endif /* KG_HOSTIF_USB_HID */
148  #if KG_HOSTIF & KG_HOSTIF_BT2_HID
149  BTMouse.scroll(hidScrollDY, 0);
150  #endif /* KG_HOSTIF_BT2_HID */
151  #if KG_HOSTIF & KG_HOSTIF_R400_HID
152  RX400.scroll(hidScrollDY);
153  #endif /* KG_HOSTIF_R400_HID */
154  DEBUG_HID_MOUSE(Serial.print("scroll "));
155  DEBUG_HID_MOUSE(Serial.println(hidScrollDY));
156  }
157 }

220 void mouse_down(uint8_t button) {
221  DEBUG_TOUCHSET(Serial.print("touchset hidMouseDown "));
222  DEBUG_TOUCHSET(Serial.println(button));
223  #if KG_HID & KG_HID_MOUSE

225  #if KG_HOSTIF & KG_HOSTIF_USB_HID
226  if (interfaceUSBHIDReady) {
227  Mouse.set_buttons((hidMouseDown & 1) > 0 ? 1 : 0, (hidMouseDown & 2) > 0 ? 1 : 0, (hidMouseDown & 4) > 0 ? 1 : 0);
228  }
229  #endif /* KG_HOSTIF_USB_HID */
230  #endif
231 }

240 void mouse_up(uint8_t button) {
241  DEBUG_TOUCHSET(Serial.print("touchset mouseup "));
242  DEBUG_TOUCHSET(Serial.println(button));
243  #if KG_HID & KG_HID_MOUSE
244  if ((hidMouseDown & button) > 0) {
245  hidMouseDown -= button;
246  #if KG_HOSTIF & KG_HOSTIF_USB_HID
247  if (interfaceUSBHIDReady) {
248  Mouse.set_buttons((hidMouseDown & 1) > 0 ? 1 : 0, (hidMouseDown & 2) > 0 ? 1 : 0, (hidMouseDown & 4) > 0 ? 1 : 0);
249  }
250  #endif /* KG_HOSTIF_USB_HID */
251  }
252  #endif
253 }
