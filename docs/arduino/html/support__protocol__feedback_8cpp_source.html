59 uint8_t process_protocol_command_feedback(uint8_t *rxPacket) {
60  // check for valid command IDs
61  uint8_t protocol_error = 0;
62  switch (rxPacket[3]) {
63  #if KG_FEEDBACK & KG_FEEDBACK_BLINK
64  case KG_PACKET_ID_CMD_FEEDBACK_GET_BLINK_MODE: // 0x01
65  // feedback_get_blink_mode()(uint8_t mode)
66  // parameters = 0 bytes
67  if (rxPacket[1] != 0) {
68  // incorrect parameter length
69  protocol_error = KG_PROTOCOL_ERROR_PARAMETER_LENGTH;
70  } else {
71  // run command
72  uint8_t mode;
73  uint16_t result = kg_cmd_feedback_get_blink_mode(&mode);
74 
75  // build response
76  uint8_t payload[1] = { mode };
77 
78  // send response
79  send_keyglove_packet(KG_PACKET_TYPE_COMMAND, 1, rxPacket[2], rxPacket[3], payload);
80  }
81  break;
82  #endif // KG_FEEDBACK & KG_FEEDBACK_BLINK
83 
84  #if KG_FEEDBACK & KG_FEEDBACK_BLINK
85  case KG_PACKET_ID_CMD_FEEDBACK_SET_BLINK_MODE: // 0x02
86  // feedback_set_blink_mode(uint8_t mode)(uint16_t result)
87  // parameters = 1 byte
88  if (rxPacket[1] != 1) {
89  // incorrect parameter length
90  protocol_error = KG_PROTOCOL_ERROR_PARAMETER_LENGTH;
91  } else {
92  // run command
93  uint16_t result = kg_cmd_feedback_set_blink_mode(rxPacket[4]);
94 
95  // build response
96  uint8_t payload[2] = { result & 0xFF, (result >> 8) & 0xFF };
97 
98  // send response
99  send_keyglove_packet(KG_PACKET_TYPE_COMMAND, 2, rxPacket[2], rxPacket[3], payload);
