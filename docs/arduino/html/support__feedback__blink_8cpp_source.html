72  // logic: mode=0 -> off, mode=5 -> on, else no immediate change

106 void update_feedback_blink() {
107  // each "keygloveTick" is 10ms, loops at 100 (1 second)
108  // each "blinkTick" is 50ms, loops at cycle period (max 12.5 seconds = 250, near 255)
109  if (feedbackBlinkLoop && (keygloveTick % 5) == 0) {
110  //feedbackBlinkMod = feedbackBlinkTick % feedbackBlinkLoop;
111 
112  if (feedbackBlinkMode == KG_BLINK_MODE_B200_100) {
113  feedback_set_blink_logic(square_wave(feedbackBlinkTick, 4, 50));
114  } else if (feedbackBlinkMode == KG_BLINK_MODE_B200_50) {
115  feedback_set_blink_logic(square_wave(feedbackBlinkTick, 2, 25));
116  } else if (feedbackBlinkMode == KG_BLINK_MODE_B1000_500) {
117  feedback_set_blink_logic(square_wave(feedbackBlinkTick, 20, 50));
118  } else if (feedbackBlinkMode == KG_BLINK_MODE_B3000_1000) {
119  feedback_set_blink_logic(square_wave(feedbackBlinkTick, 60, 33));
120  } else if (feedbackBlinkMode == KG_BLINK_MODE_B1000_100 || feedbackBlinkMode == KG_BLINK_MODE_B3000_100) {
121  feedback_set_blink_logic(feedbackBlinkTick < 2 ? square_wave(feedbackBlinkTick, 4, 50) : 0);
122  } else if (feedbackBlinkMode == KG_BLINK_MODE_B1000_100_2X || feedbackBlinkMode == KG_BLINK_MODE_B3000_100_2X) {
123  feedback_set_blink_logic(feedbackBlinkTick < 6 ? square_wave(feedbackBlinkTick, 4, 50) : 0);
124  } else if (feedbackBlinkMode == KG_BLINK_MODE_B1000_100_3X || feedbackBlinkMode == KG_BLINK_MODE_B3000_100_3X) {
125  feedback_set_blink_logic(feedbackBlinkTick < 10 ? square_wave(feedbackBlinkTick, 4, 50) : 0);
126  }
127  feedbackBlinkTick++;
128  if (feedbackBlinkTick == feedbackBlinkLoop) feedbackBlinkTick = 0;
129  }
130 }

151 uint16_t kg_cmd_feedback_set_blink_mode(uint8_t mode) {
152  if (mode >= KG_BLINK_MODE_MAX) {
153  return KG_PROTOCOL_ERROR_PARAMETER_RANGE;
154  } else {
155  feedback_set_blink_mode((feedback_blink_mode_t)mode);
156 
157  // send kg_evt_feedback_blink_mode packet (if we aren't setting it from an API command)
158  if (!inBinPacket) {
159  uint8_t payload[1] = { mode };
160  skipPacket = 0;
161  if (kg_evt_feedback_blink_mode) skipPacket = kg_evt_feedback_blink_mode(mode);
162  if (!skipPacket) send_keyglove_packet(KG_PACKET_TYPE_EVENT, 1, KG_PACKET_CLASS_FEEDBACK, KG_PACKET_ID_EVT_FEEDBACK_BLINK_MODE, payload);
163  }
164  }
165  return 0; // success
