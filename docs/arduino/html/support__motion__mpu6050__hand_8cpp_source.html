146 void update_motion_mpu6050_hand() {
147  uint8_t mpuInt = mpuHand.getIntStatus();
148  if (mpuInt & 0x01) {
149  // read raw motion data
150  mpuHand.getMotion6(&aaRaw.x, &aaRaw.y, &aaRaw.z, &gvRaw.x, &gvRaw.y, &gvRaw.z);
151 
152  // store previous accel/gyro values
153  aa0.x = aa.x;
154  aa0.y = aa.y;
155  aa0.z = aa.z;
156  gv0.x = gv.x;
157  gv0.y = gv.y;
158  gv0.z = gv.z;
159 
160  // simple smoothing filter
161  aa.x = aa0.x + (0.25 * (aaRaw.x - aa0.x));
162  aa.y = aa0.y + (0.25 * (aaRaw.y - aa0.y));
163  aa.z = aa0.z + (0.25 * (aaRaw.z - aa0.z));
164  gv.x = gv0.x + (0.25 * (gvRaw.x - gv0.x));
165  gv.y = gv0.y + (0.25 * (gvRaw.y - gv0.y));
166  gv.z = gv0.z + (0.25 * (gvRaw.z - gv0.z));
167 
168  // build and send kg_evt_motion_data packet
169  uint8_t payload[15];
170  payload[0] = 0x00; // sensor 0
171  payload[1] = 0x03; // 1=accel, 2=gyro, 1|2 = 0x03
172  payload[2] = 0x0C; // 12 bytes of motion data (6 axes, 2 bytes each)
173  payload[3] = aa.x & 0xFF;
174  payload[4] = aa.x >> 8;
175  payload[5] = aa.y & 0xFF;
176  payload[6] = aa.y >> 8;
177  payload[7] = aa.z & 0xFF;
178  payload[8] = aa.z >> 8;
179  payload[9] = gv.x & 0xFF;
180  payload[10] = gv.x >> 8;
181  payload[11] = gv.y & 0xFF;
182  payload[12] = gv.y >> 8;
183  payload[13] = gv.z & 0xFF;
184  payload[14] = gv.z >> 8;
185  skipPacket = 0;
186  if (kg_evt_motion_data) skipPacket = kg_evt_motion_data(payload[0], payload[1], payload[2], payload + 3);
187  if (!skipPacket) send_keyglove_packet(KG_PACKET_TYPE_EVENT, sizeof(payload), KG_PACKET_CLASS_MOTION, KG_PACKET_ID_EVT_MOTION_DATA, payload);
188  }
189  if (mpuInt & 0x20) {
190  send_keyglove_log(KG_LOG_LEVEL_VERBOSE, 10, F("MOTION INT"));
191  } else if (mpuInt & 0x40) {
192  send_keyglove_log(KG_LOG_LEVEL_VERBOSE, 10, F("ZEROMO INT"));
193  }
194 }
