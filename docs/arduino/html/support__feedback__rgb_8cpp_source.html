105 void feedback_set_rgb_mode(feedback_rgb_mode_t r, feedback_rgb_mode_t g, feedback_rgb_mode_t b) {
106  feedbackRGBMode[0] = r;
107  feedbackRGBMode[1] = g;
108  feedbackRGBMode[2] = b;
109  feedbackRGBTick[0] = feedbackRGBTick[1] = feedbackRGBTick[2] = 0;
110  feedbackRGBLoop[0] = feedbackRGBLoop[1] = feedbackRGBLoop[2] = 0;
111 
112  for (uint8_t i = 0; i < 3; i++) {
113  if (feedbackRGBMode[i] == KG_RGB_MODE_B200_100 || feedbackRGBMode[i] == KG_RGB_MODE_F200_100 ||
114  feedbackRGBMode[i] == KG_RGB_MODE_B200_50 || feedbackRGBMode[i] == KG_RGB_MODE_F200_50
115  ) feedbackRGBLoop[i] = 20;
116  else if (feedbackRGBMode[i] == KG_RGB_MODE_B1000_500 || feedbackRGBMode[i] == KG_RGB_MODE_F1000_500 ||
117  feedbackRGBMode[i] == KG_RGB_MODE_B1000_100 || feedbackRGBMode[i] == KG_RGB_MODE_F1000_100 ||
118  feedbackRGBMode[i] == KG_RGB_MODE_B1000_100_2X || feedbackRGBMode[i] == KG_RGB_MODE_F1000_100_2X ||
119  feedbackRGBMode[i] == KG_RGB_MODE_B1000_100_3X || feedbackRGBMode[i] == KG_RGB_MODE_F1000_100_3X ||
120  feedbackRGBMode[i] == KG_RGB_MODE_F1000_1000
121  ) feedbackRGBLoop[i] = 100;
122  else if (feedbackRGBMode[i] == KG_RGB_MODE_B3000_1000 || feedbackRGBMode[i] == KG_RGB_MODE_F3000_1000 ||
123  feedbackRGBMode[i] == KG_RGB_MODE_B3000_100 || feedbackRGBMode[i] == KG_RGB_MODE_F3000_100 ||
124  feedbackRGBMode[i] == KG_RGB_MODE_B3000_100_2X || feedbackRGBMode[i] == KG_RGB_MODE_F3000_100_2X ||
125  feedbackRGBMode[i] == KG_RGB_MODE_B3000_100_3X || feedbackRGBMode[i] == KG_RGB_MODE_F3000_100_3X ||
126  feedbackRGBMode[i] == KG_RGB_MODE_F3000_3000
127  ) feedbackRGBLoop[i] = 300;
128  }
129 
130  // logic: r=OFF -> off, r=ON -> on, else no immediate change, update_feedback_rgb() will handling it
131  feedback_set_rgb_digital(
132  r == KG_RGB_MODE_OFF ? 0 : (r == KG_RGB_MODE_SOLID ? 1 : 255),
133  g == KG_RGB_MODE_OFF ? 0 : (g == KG_RGB_MODE_SOLID ? 1 : 255),
134  b == KG_RGB_MODE_OFF ? 0 : (b == KG_RGB_MODE_SOLID ? 1 : 255)
135  );
136 }

158 void update_feedback_rgb() {
159  // each "keygloveTick" is 10ms, loops at 100 (1 second)
160  // each "RGBTick" is also 10ms, loops at cycle period (can be greater than 1 second, actually 6553 seconds)
161  for (uint8_t i = 0; i < 3; i++) {
162  if (feedbackRGBMode[i] == KG_RGB_MODE_B200_100) {
163  feedback_set_rgb_digital_channel(i, square_wave(feedbackRGBTick[i], 20, 50));
164  } else if (feedbackRGBMode[i] == KG_RGB_MODE_B200_50) {
165  feedback_set_rgb_digital_channel(i, square_wave(feedbackRGBTick[i], 10, 25));
166  } else if (feedbackRGBMode[i] == KG_RGB_MODE_B1000_500) {
167  feedback_set_rgb_digital_channel(i, square_wave(feedbackRGBTick[i], 100, 50));
168  } else if (feedbackRGBMode[i] == KG_RGB_MODE_B3000_1000) {
169  feedback_set_rgb_digital_channel(i, square_wave(feedbackRGBTick[i], 300, 33));
170  } else if (feedbackRGBMode[i] == KG_RGB_MODE_B1000_100 || feedbackRGBMode[i] == KG_RGB_MODE_B3000_100) {
171  feedback_set_rgb_digital_channel(i, feedbackRGBTick[i] < 10 ? square_wave(feedbackRGBTick[i], 20, 50) : 0);
172  } else if (feedbackRGBMode[i] == KG_RGB_MODE_B1000_100_2X || feedbackRGBMode[i] == KG_RGB_MODE_B3000_100_2X) {
173  feedback_set_rgb_digital_channel(i, feedbackRGBTick[i] < 30 ? square_wave(feedbackRGBTick[i], 20, 50) : 0);
174  } else if (feedbackRGBMode[i] == KG_RGB_MODE_B1000_100_3X || feedbackRGBMode[i] == KG_RGB_MODE_B3000_100_3X) {
175  feedback_set_rgb_digital_channel(i, feedbackRGBTick[i] < 50 ? square_wave(feedbackRGBTick[i], 20, 50) : 0);
176 
177  } else if (feedbackRGBMode[i] == KG_RGB_MODE_F200_50) {
178  feedback_set_rgb_analog_channel(i, feedbackRGBTick[i] < 5 ? triangle_wave(feedbackRGBTick[i], 10, 400) : 0);
179  } else if (feedbackRGBMode[i] == KG_RGB_MODE_F1000_500) {
180  feedback_set_rgb_analog_channel(i, feedbackRGBTick[i] < 50 ? triangle_wave(feedbackRGBTick[i], 100, 400) : 0);
181  } else if (feedbackRGBMode[i] == KG_RGB_MODE_F1000_1000) {
182  feedback_set_rgb_analog_channel(i, triangle_wave(feedbackRGBTick[i], 200, 400));
183  } else if (feedbackRGBMode[i] == KG_RGB_MODE_F3000_1000) {
184  feedback_set_rgb_analog_channel(i, feedbackRGBTick[i] < 100 ? triangle_wave(feedbackRGBTick[i], 200, 400) : 0);
185  } else if (feedbackRGBMode[i] == KG_RGB_MODE_F3000_3000) {
186  feedback_set_rgb_analog_channel(i, triangle_wave(feedbackRGBTick[i], 600, 400));
187  } else if (feedbackRGBMode[i] == KG_RGB_MODE_F200_100 || feedbackRGBMode[i] == KG_RGB_MODE_F1000_100 || feedbackRGBMode[i] == KG_RGB_MODE_F3000_100) {
188  feedback_set_rgb_analog_channel(i, feedbackRGBTick[i] < 10 ? triangle_wave(feedbackRGBTick[i], 20, 400) : 0);
189  } else if (feedbackRGBMode[i] == KG_RGB_MODE_F1000_100_2X || feedbackRGBMode[i] == KG_RGB_MODE_F3000_100_2X) {
190  feedback_set_rgb_analog_channel(i, feedbackRGBTick[i] < 30 ? triangle_wave(feedbackRGBTick[i], 20, 400) : 0);
191  } else if (feedbackRGBMode[i] == KG_RGB_MODE_F1000_100_3X || feedbackRGBMode[i] == KG_RGB_MODE_F3000_100_3X) {
192  feedback_set_rgb_analog_channel(i, feedbackRGBTick[i] < 50 ? triangle_wave(feedbackRGBTick[i], 20, 400) : 0);
193  }
194 
195  feedbackRGBTick[i]++;
196  if (feedbackRGBTick[i] == feedbackRGBLoop[i]) feedbackRGBTick[i] = 0;
197  }
198 }

228 uint16_t kg_cmd_feedback_set_rgb_mode(uint8_t index, uint8_t mode_red, uint8_t mode_green, uint8_t mode_blue) {
229  if (mode_red >= KG_RGB_MODE_MAX || mode_green >= KG_RGB_MODE_MAX || mode_blue >= KG_RGB_MODE_MAX) {
230  return KG_PROTOCOL_ERROR_PARAMETER_RANGE;
231  } else {
232  // "index" is currently ignored, as there is only one RGB device in the design
233  feedback_set_rgb_mode((feedback_rgb_mode_t)mode_red, (feedback_rgb_mode_t)mode_green, (feedback_rgb_mode_t)mode_blue);
234 
235  // send kg_evt_feedback_rgb_mode packet (if we aren't setting it from an API command)
236  if (!inBinPacket) {
237  uint8_t payload[4] = { index, mode_red, mode_green, mode_blue };
238  skipPacket = 0;
239  if (kg_evt_feedback_rgb_mode) skipPacket = kg_evt_feedback_rgb_mode(index, mode_red, mode_green, mode_blue);
240  if (!skipPacket) send_keyglove_packet(KG_PACKET_TYPE_EVENT, 4, KG_PACKET_CLASS_FEEDBACK, KG_PACKET_ID_EVT_FEEDBACK_RGB_MODE, payload);
241  }
242  }
243  return 0; // success
244 }
