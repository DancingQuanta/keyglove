97 void protocol_parse(uint8_t inputByte) {
98  if (rxPacketLength + 1 == rxPacketSize) {
99  rxPacketSize += 32;
100  rxPacket = (uint8_t *)realloc(rxPacket, rxPacketSize);
101  if (!rxPacket) {
102  // failed to reallocate to new chunk of memory
103  // this should NEVER happen, but if it does, I want to know
104  uint8_t payload[2] = { KG_PROTOCOL_ERROR_NULL_POINTER & 0xFF, KG_PROTOCOL_ERROR_NULL_POINTER >> 8 };
105  skipPacket = 0;
106  if (kg_evt_protocol_error) skipPacket = kg_evt_protocol_error(KG_PROTOCOL_ERROR_NULL_POINTER);
107  if (!skipPacket) send_keyglove_packet(KG_PACKET_TYPE_EVENT, 2, KG_PACKET_CLASS_PROTOCOL, KG_PACKET_ID_EVT_PROTOCOL_ERROR, payload);
108  return;
109  }
110  }
111  if (!inBinPacket && inputByte == KG_PACKET_TYPE_COMMAND) { // "bait" byte, 0xC0
112  packetStartTime = millis();
113  inBinPacket = true;
114  rxPacket[0] = inputByte;
115  rxPacketLength = 1; // initialize buffer length to include only 1st header byte (so far)
116  binDataLength = 0;
117  } else if (inBinPacket && rxPacketLength == 1) { // data length byte
118  binDataLength = inputByte;
119  if (binDataLength > 250) {
120  // error (data payload too long)
121  uint8_t payload[2] = { KG_PROTOCOL_ERROR_BAD_LENGTH, 0x00 };
122  skipPacket = 0;
123  if (kg_evt_protocol_error) skipPacket = kg_evt_protocol_error(payload[0]);
124  if (!skipPacket) send_keyglove_packet(KG_PACKET_TYPE_EVENT, 2, KG_PACKET_CLASS_PROTOCOL, KG_PACKET_ID_EVT_PROTOCOL_ERROR, payload);
125  inBinPacket = false;
126  rxPacketLength = 0;
127  } else {
128  rxPacket[rxPacketLength++] = inputByte;
129  }
130  } else {
131  rxPacket[rxPacketLength++] = inputByte;
132  if (inBinPacket && rxPacketLength - 4 == binDataLength) {
133  // process packet that just came in, passing to appropriate main handler
134  uint8_t protocol_error = 0;
135 
136  // filter incoming packets for custom behavior
137  if (filter_incoming_keyglove_packet(rxPacket) == 0) {
138  switch (rxPacket[2]) {
139  //case KG_PACKET_CLASS_PROTOCOL:
140 
141  case KG_PACKET_CLASS_SYSTEM:
142  protocol_error = process_protocol_command_system(rxPacket);
143  break;
144 
145  case KG_PACKET_CLASS_TOUCH:
146  protocol_error = process_protocol_command_touch(rxPacket);
147  break;
148 
149  #if KG_MOTION > 0
150  case KG_PACKET_CLASS_MOTION:
151  protocol_error = process_protocol_command_motion(rxPacket);
152  break;
153  #endif
154 
155  #if KG_FEEDBACK > 0
156  case KG_PACKET_CLASS_FEEDBACK:
157  protocol_error = process_protocol_command_feedback(rxPacket);
158  break;
159  #endif
160 
161  #if KG_FLEX > 0
162  case KG_PACKET_CLASS_FLEX:
163  protocol_error = process_protocol_command_flex(rxPacket);
164  break;
165  #endif
166 
167  #if KG_PRESSURE > 0
168  case KG_PACKET_CLASS_PRESSURE:
169  protocol_error = process_protocol_command_pressure(rxPacket);
170  break;
171  #endif
172 
173  #if KG_TOUCHSET > 0
174  case KG_PACKET_CLASS_TOUCHSET:
175  protocol_error = process_protocol_command_touchset(rxPacket);
176  break;
177  #endif
178 
179  #if (KG_HOSTIF & HG_HOSTIF_BT2_SPP) || (KG_HOSTIF & KG_HOSTIF_BT2_RAWHID) || (KG_HOSTIF & KG_HOSTIF_BT2_IAP)
180  case KG_PACKET_CLASS_BLUETOOTH:
181  protocol_error = process_protocol_command_bluetooth(rxPacket);
182  break;
183  #endif
184 
185  default:
186  // check for custom protocol, will return KG_PROTOCOL_ERROR_INVALID_COMMAND if no matches
187  protocol_error = KG_PROTOCOL_ERROR_INVALID_COMMAND;
188  break;
189  }
190 
191  // check for errors (e.g. unhandled, bad arguments, etc.)
192  if (protocol_error) {
193  // check for custom protocol, will return KG_PROTOCOL_ERROR_INVALID_COMMAND if no matches
194  protocol_error = process_protocol_command_custom(rxPacket);
195 
196  // if we still have an error, then there is no custom functionality implemented for this class/ID combination
197  if (protocol_error) {
198  uint8_t payload[2] = { protocol_error, 0x00 };
199  skipPacket = 0;
200  if (kg_evt_protocol_error) skipPacket = kg_evt_protocol_error(payload[0]);
201  if (!skipPacket) send_keyglove_packet(KG_PACKET_TYPE_EVENT, 2, KG_PACKET_CLASS_PROTOCOL, KG_PACKET_ID_EVT_PROTOCOL_ERROR, payload);
202  }
203  }
204  }
205 
206  // reset packet status/length
207  reset_keyglove_rx_packet();
208  }
209  }
210 }

216 uint8_t check_incoming_protocol_data() {
217  #if KG_HOSTIF & KG_HOSTIF_USB_SERIAL
218  // read available data from USB virtual serial
219  if (interfaceUSBSerialReady && (interfaceUSBSerialMode & KG_INTERFACE_MODE_INCOMING_PACKET) != 0) {
220  while ((rxByte = USBSerial.read()) < 256) {
221  lastCommandInterfaceNum = KG_INTERFACENUM_USB_SERIAL;
222  protocol_parse((uint8_t)rxByte);
223  }
224  }
225  #endif
226 
227  #if KG_HOSTIF & KG_HOSTIF_USB_RAWHID
228  // read raw HID data over USB
229  if (interfaceUSBRawHIDReady && (interfaceUSBRawHIDMode & KG_INTERFACE_MODE_INCOMING_PACKET) != 0) {
230  int8_t bytes = RawHID.recv(rxRawHIDPacket, 0);
231  if (bytes > 0) {
232  lastCommandInterfaceNum = KG_INTERFACENUM_USB_RAWHID;
233  for (int8_t i = 0; i < rxRawHIDPacket[0]; i++) protocol_parse(rxRawHIDPacket[i + 1]);
234  }
235  }
236  #endif
237 
238  #if (KG_HOSTIF & HG_HOSTIF_BT2_SPP) || (KG_HOSTIF & KG_HOSTIF_BT2_HID) || (KG_HOSTIF & KG_HOSTIF_BT2_RAWHID) || (KG_HOSTIF & KG_HOSTIF_BT2_IAP)
239  // see if there is any data to deal with from Bluetooth
240  bluetooth_check_incoming_protocol_data();
241  #endif
242 
243  // check for protocol timeout condition
244  if (inBinPacket && (millis() - packetStartTime) > KG_PROTOCOL_RX_TIMEOUT) {
245  // error (data payload too long)
246  uint8_t payload[2] = { KG_PROTOCOL_ERROR_PACKET_TIMEOUT, 0x00 };
247  skipPacket = 0;
248  if (kg_evt_protocol_error) skipPacket = kg_evt_protocol_error(payload[0]);
249  if (!skipPacket) send_keyglove_packet(KG_PACKET_TYPE_EVENT, 2, KG_PACKET_CLASS_PROTOCOL, KG_PACKET_ID_EVT_PROTOCOL_ERROR, payload);
250  inBinPacket = false;
251  rxPacketLength = 0;
252  }
253 
254  return 0;
255 }

294 uint8_t queue_keyglove_packet(uint8_t packetType, uint8_t payloadLength, uint8_t packetClass, uint8_t packetId, uint8_t *payload) {
295  uint8_t packetLength = payloadLength + 4;
296  if (txQueueSize == 0) {
297  if (!(txQueue = (uint8_t *)malloc(384))) {
298  return 1; // couldn't allocate block of memory
299  }
300  txQueueSize = 384;
301  } else if (txQueueSize + packetLength > txQueueSize) {
302  if (!(txQueue = (uint8_t *)realloc(txQueue, txQueueSize + packetLength + 64))) {
303  return 2; // couldn't re-allocate bigger block of memory
304  }
305  txQueueSize += packetLength + 64;
306  }
307  txQueue[txQueueLength++] = packetType;
308  txQueue[txQueueLength++] = payloadLength;
309  txQueue[txQueueLength++] = packetClass;
310  txQueue[txQueueLength++] = packetId;
311  if (payloadLength) {
312  memcpy(txQueue + txQueueLength, payload, payloadLength);
313  txQueueLength += payloadLength;
314  }
315  return 0;
316 }

327 uint8_t send_keyglove_packet(uint8_t packetType, uint8_t payloadLength, uint8_t packetClass, uint8_t packetId, uint8_t *payload) {
328  // validate payload length
329  if ((payload == NULL && payloadLength > 0) || payloadLength > 250) {
330  // payload specified but not provided, or too long
331  return 1;
332  }
333 
334  // filter outgoing packets for custom behavior
335  if (filter_outgoing_keyglove_packet(&packetType, &payloadLength, &packetClass, &packetId, payload)) return 255;
336 
337  // allocate and check full packet buffer
338  uint8_t *buffer = (uint8_t *)malloc(4 + payloadLength);
339  if (buffer == 0) {
340  // couldn't allocate packet buffer...uh oh
341  return 2;
342  }
343 
344  // certain outgoing packets should only be sent on one specific interface, the last one which was used
345  uint8_t specificInterface = (packetType != KG_PACKET_TYPE_EVENT || packetClass == KG_PACKET_CLASS_PROTOCOL);
346 
347  buffer[0] = packetType;
348  buffer[1] = payloadLength;
349  buffer[2] = packetClass;
350  buffer[3] = packetId;
351  if (payloadLength) memcpy(buffer + 4, payload, payloadLength);
352  uint8_t length = 4 + payloadLength;
353 
354  #if KG_HOSTIF & KG_HOSTIF_USB_SERIAL
355  if (!specificInterface || lastCommandInterfaceNum == KG_INTERFACENUM_USB_SERIAL) {
356  // send packet out over wired serial (USB virtual serial)
357  if (interfaceUSBSerialReady && (interfaceUSBSerialMode & KG_INTERFACE_MODE_OUTGOING_PACKET) != 0) {
358  USBSerial.write((const uint8_t *)buffer, length); // packet data
359  }
360  }
361  #endif
362 
363  #if KG_HOSTIF & KG_HOSTIF_USB_RAWHID
364  if (!specificInterface || lastCommandInterfaceNum == KG_INTERFACENUM_USB_RAWHID) {
365  // send packet out over wired custom HID interface (USB raw HID)
366  // 64-byte packets, formatted where byte 0 is [0-64] and bytes 1-63 are data
367  if (interfaceUSBRawHIDReady && (interfaceUSBRawHIDMode & KG_INTERFACE_MODE_OUTGOING_PACKET) != 0) {
368  int8_t bytes;
369  for (uint8_t i = 0; i < length; i += (USB_RAWHID_TX_SIZE - 1)) {
370  memset(txRawHIDPacket, 0, USB_RAWHID_TX_SIZE);
371  txRawHIDPacket[0] = min(USB_RAWHID_TX_SIZE - 1, length - i);
372  for (uint8_t j = 0; j < (USB_RAWHID_TX_SIZE - 1); j++) {
373  if (i + j >= length) break;
374  txRawHIDPacket[j + 1] = buffer[i + j];
375  }
376  bytes = RawHID.send(txRawHIDPacket, 2);
377  }
378  }
379  }
380  #endif
381 
382  #if (KG_HOSTIF & HG_HOSTIF_BT2_SPP) || (KG_HOSTIF & KG_HOSTIF_BT2_HID) || (KG_HOSTIF & KG_HOSTIF_BT2_RAWHID) || (KG_HOSTIF & KG_HOSTIF_BT2_IAP)
383  // send packet via Bluetooth if necessary
384  bluetooth_send_keyglove_packet_buffer(buffer, length, specificInterface);
385  #endif
386 
387  // KG_HID_KEYBOARD and KG_HID_MOUSE are handled elsewhere and deal with other kinds of data
388 
389  // tidy up!
390  free(buffer);
391 
392  return 0;
393 }

400 uint16_t send_keyglove_queue() {
401  // check for queued packets
402  if (txQueueLength) {
403  send_keyglove_packet(txQueue[0], txQueue[1], txQueue[2], txQueue[3], txQueue + 4);
404  if (txQueueLength > txQueue[1] + 4) {
405  // queue has more left, so move it into position
406  txQueueLength -= (txQueue[1] + 4);
407  memmove(txQueue, txQueue + (txQueue[1] + 4), txQueueLength);
408  } else {
409  // queue is empty, free up the memory
410  txQueueLength = 0;
411  txQueueSize = 0;
412  free(txQueue);
413  }
414  }
415  return txQueueLength;
416 }
