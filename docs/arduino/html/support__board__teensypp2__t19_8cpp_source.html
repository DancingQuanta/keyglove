150 void setup_board() {
151  // setup internal 100Hz "tick" interrupt
152  // thanks to http://www.arduino.cc/cgi-bin/yabb2/YaBB.pl?num=1212098919 (and 'bens')
153  // also, lots of timer info here and here:
154  // http://www.avrfreaks.net/index.php?name=PNphpBB2&file=viewtopic&t=50106
155  // http://www.avrbeginners.net/architecture/timers/timers.html
156  // also, section 15.10 of the AT90USB128x datasheet:
157  // http://www.atmel.com/Images/doc7593.pdf
158 
159  // set up Timer1
160  // WGM13 = 0 \
161  // WGM12 = 1 | --> CTC mode, TOP=OCR1A
162  // WGM11 = 0 |
163  // WGM10 = 0 /
164  // CS12 = 1 \
165  // CS11 = 0 | --> clk/256 prescaler
166  // CS10 = 0 /
167 
168  TCCR1A = 0x00; // TCCR1A: COM1A1=0, COM1A0=0, COM1B1=0, COM1B0=0, COM1C1=0, COM1C0=0, WGM11=0, WGM10=0
169  TCCR1B = 0x0A; // TCCR1B: ICNC1=0, ICES1=0, -, WGM13=1, WGM12=1, CS12=1, CS11=0, CS10=0
170  OCR1A = 0x2710; // 10ms interval @ 8MHz
171  //OCR1A = 0x0500; // speed up ALL THE THINGS!!
172  TIMSK1 |= (1 << OCIE1A); // enable TIMER1 output compare match interrupt
173 
174  // set up pin change interrupts on relevant Port B pins
175  PCICR = 0x01; // PCICR: PCIE0=1 (enable pin change interrupts)
176  PCMSK0 = 0x07; // PCMSK0: PCINT0/1/2=1 (enable PB0/PB1/PB2 pins)
177 
178  /*
179  __|||||__
180  GND | USB | VCC
181  1 PB7 27 | | 26 PB6 Y
182  SCL PD0 0 | | 25 PB5 8
183  SDA PD1 1 | | 24 PB4 SPK
184  RXD PD2 2 | | 23 PB3 VIB
185  TXD PD3 3 | | 22 PB2 CS2 (STAT2)
186  RTS PD4 4 | 37.36 | 21 PB1 CS1 (STAT1)
187  7 PD5 5 | | 20 PB0 CS0 (/PG)
188  LED PD6 6 | | 19 PE7 CTS (INT7)
189  L PD7 7 | | 18 PE6 MPU (INT6)
190  K PE0 8 | | GND
191  J PE1 9 | | AREF
192  6 PC0 10 | | 38 PF0 BAT (VBDIV)
193  I PC1 11 | 32 . 28 | 39 PF1 A
194  H PC2 12 | 33 . 29 | 40 PF2 B
195  G PC3 13 | 34 . 30 | 41 PF3 C
196  RED PC4 14 | 35 . 31 | 42 PF4 4
197  GRN PC5 15 | | 43 PF5 D
198  BLU PC6 16 | | 44 PF6 E
199  5 PC7 17 |_________| 45 PF7 F
200  */
201 
202  // setup touch/battery sensors and make sure we enable internal pullup resistors EXCEPT on PF0 (BAT)
203  DDRB &= (0x18); // 7,6,5, , ,2,1,0
204  PORTB |= ~(0x18);
205  DDRC &= (0x70); // 7, , , ,3,2,1,0
206  PORTC |= ~(0x70);
207  DDRD &= (0x5F); // 7, ,5, , , , ,
208  PORTD |= ~(0x5F);
209  DDRE &= (0xFC); // , , , , , ,1,0
210  PORTE |= ~(0xFC);
211  DDRF &= (0x00); // 0,1,2,3,4,5,6,7
212  PORTF = ~(0x01);
213 
214  // set UART RTS to be output/low (asserted, i.e. "ready to receive")
215  DDRD |= 0x10;
216  PORTD &= ~(0x10);
217 
218  /*
219  uint8_t b = PINB;
220  int16_t bat = analogRead(0);
221  Serial.println(b & 0x07);
222  Serial.println(bat);
223  // 4.2v in with USB supply, ~880 reading with 47k/100k voltage divider
224  // http://www.raltron.com/cust/tools/voltage_divider.asp
225  */
226 
227  /*
228  Relevant pin change interrupts for touch sensors:
229  - 1, 27, PB7 (palm)
230  - Y, 26, PB6 (thumbtip)
231  - 8, 25, PB5 (thumbnail)
232  */
233 
234  #if KG_HOSTIF & KG_HOSTIF_USB_SERIAL
235  // start USB serial interface
236  USBSerial.begin(KG_HOSTIF_USB_SERIAL_BAUD);
237  interfaceUSBSerialReady = true;
238  interfaceUSBSerialMode = KG_APIMODE_USB_SERIAL;
239  #endif
240 
241  #if KG_HOSTIF & KG_HOSTIF_USB_RAWHID
242  interfaceUSBRawHIDReady = true;
243  interfaceUSBRawHIDMode = KG_APIMODE_USB_RAWHID;
244  #endif
245 
246  #if KG_HOSTIF & KG_HOSTIF_USB_HID
247  interfaceUSBHIDReady = true;
248  #endif
249 
250  #if KG_HOSTIF & (KG_HOSTIF_BT2_SERIAL | KG_HOSTIF_BT2_RAWHID | KG_HOSTIF_BT2_HID | KG_HOSTIF_BT2_IAP)
251  // start BT2 serial interface if any BT2 interfaces are used
252  BT2Serial.begin(KG_HOSTIF_BT2_SERIAL_BAUD);
253  #endif
254 }

267 void update_board_touch(uint8_t *touches) {
268  /*
269  __|||||__
270  GND | USB | VCC
271  1 PB7 27 | | 26 PB6 Y
272  SCL PD0 0 | | 25 PB5 8
273  SDA PD1 1 | | 24 PB4 SPK
274  RXD PD2 2 | | 23 PB3 VIB
275  TXD PD3 3 | | 22 PB2 CS2 (STAT2)
276  RTS PD4 4 | 37.36 | 21 PB1 CS1 (STAT1)
277  7 PD5 5 | | 20 PB0 CS0 (/PG)
278  LED PD6 6 | | 19 PE7 CTS (INT7)
279  L PD7 7 | | 18 PE6 MPU (INT6)
280  K PE0 8 | | GND
281  J PE1 9 | | AREF
282  6 PC0 10 | | 38 PF0 BAT (VBDIV)
283  I PC1 11 | 32 . 28 | 39 PF1 A
284  H PC2 12 | 33 . 29 | 40 PF2 B
285  G PC3 13 | 34 . 30 | 41 PF3 C
286  RED PC4 14 | 35 . 31 | 42 PF4 4
287  GRN PC5 15 | | 43 PF5 D
288  BLU PC6 16 | | 44 PF6 E
289  5 PC7 17 |_________| 45 PF7 F
290  */
291 
292  // check on Y combinations (PB6)
293  SET(DDRB, 6); // set to OUTPUT
294  CLR(PORTB, 6); // set to LOW
295  delayMicroseconds(3); // give the poor receiving pins a chance to change state
296  _pinb = PINB; _pinc = PINC; _pind = PIND; _pine = PINE; _pinf = PINF;
297  CLR(DDRB, 6); // set to INPUT
298  SET(PORTB, 6); // pull HIGH
299  if (!(_pinf & (1 << 1))) touches[0] |= 0x01; // A (PF1)
300  if (!(_pinf & (1 << 2))) touches[0] |= 0x02; // B (PF2)
301  if (!(_pinf & (1 << 3))) touches[0] |= 0x04; // C (PF3)
302  if (!(_pinf & (1 << 5))) touches[0] |= 0x08; // D (PF5)
303  if (!(_pinf & (1 << 6))) touches[0] |= 0x10; // E (PF6)
304  if (!(_pinf & (1 << 7))) touches[0] |= 0x20; // F (PF7)
305  if (!(_pinc & (1 << 3))) touches[0] |= 0x40; // G (PC3)
306  if (!(_pinc & (1 << 2))) touches[0] |= 0x80; // H (PC2)
307  if (!(_pinc & (1 << 1))) touches[1] |= 0x01; // I (PC1)
308  if (!(_pine & (1 << 1))) touches[1] |= 0x02; // J (PE1)
309  if (!(_pine & (1 << 0))) touches[1] |= 0x04; // K (PE0)
310  if (!(_pind & (1 << 7))) touches[1] |= 0x08; // L (PD7)
311  if (!(_pinf & (1 << 4))) touches[1] |= 0x10; // 4 (PF4)
312  if (!(_pinc & (1 << 7))) touches[1] |= 0x20; // 5 (PC7)
313  if (!(_pinc & (1 << 0))) touches[1] |= 0x40; // 6 (PC0)
314  if (!(_pind & (1 << 5))) touches[1] |= 0x80; // 7 (PD5)
315  if (!(_pinb & (1 << 7))) touches[2] |= 0x01; // 1 (PB7)
316 
317  // check on 1 combinations (PB7)
318  SET(DDRB, 7); // set to OUTPUT
319  CLR(PORTB, 7); // set to LOW
320  delayMicroseconds(3); // give the poor receiving pins a chance to change state
321  _pinc = PINC; _pine = PINE; _pinf = PINF;
322  CLR(DDRB, 7); // set to INPUT
323  SET(PORTB, 7); // pull HIGH
324  if (!(_pinf & (1 << 1))) touches[2] |= 0x02; // A (PF1)
325  if (!(_pinf & (1 << 5))) touches[2] |= 0x04; // D (PF5)
326  if (!(_pinc & (1 << 3))) touches[2] |= 0x08; // G (PC3)
327  if (!(_pine & (1 << 1))) touches[2] |= 0x10; // J (PE1)
328 
329  // check on 8 combinations (PB5)
330  SET(DDRB, 5); // set to OUTPUT
331  CLR(PORTB, 5); // set to LOW
332  delayMicroseconds(3); // give the poor receiving pins a chance to change state
333  _pinc = PINC; _pine = PINE; _pinf = PINF;
334  CLR(DDRB, 5); // set to INPUT
335  SET(PORTB, 5); // pull HIGH
336  if (!(_pinf & (1 << 1))) touches[2] |= 0x20; // A (PF1)
337  if (!(_pinf & (1 << 5))) touches[2] |= 0x40; // D (PF5)
338  if (!(_pinc & (1 << 3))) touches[2] |= 0x80; // G (PC3)
339  if (!(_pine & (1 << 1))) touches[3] |= 0x01; // J (PE1)
340 }
